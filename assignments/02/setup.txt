1.) download ubuntu 18.04 server from https://ubuntu.com/download/server
A:) Qemu
	2.) create a virtual disk of size 10G called QEMU.img
		qemu-img create QEMU.img 10G
	3.) install ubuntu form iso to QEMU.img
		qemu-system-x86_64 -smp $(nproc) -m 4096 -boot d -cdrom ubuntu-18.04.6-live-server-amd64.iso -hda QEMU.img 
	4.) after installation is completed, exit Qemu
	5.) start Qemu with all available cores 
		qemu-system-x86_64 -smp $(nproc) -m 4096 -hda QEMU.img -net nic -net user,hostfwd=tcp::2222-:22 
	6.) access the VM via ssh
		ssh user@localhost -p 2222
B:) KVM
	2.) create a virtual disk of size 10G called KVM.img
		qemu-img create KVM.img 10G
	3.) install ubuntu form iso to KVM.img
		qemu-system-x86_64 -smp $(nproc) -m 4096 -boot d -cdrom ubuntu-18.04.6-live-server-amd64.iso -hda KVM.img 
	4.) after installation is completed, exit Qemu
	5.) start Qemu with all available cores 
    	qemu-system-x86_64 -smp $(nproc) -m 4096 -hda KVM.img -net nic -net user,hostfwd=tcp::2223-:22 -enable-kvm
	6.) access the VM via ssh
		ssh user@localhost -p 2223

software:
	qemu 6.1.0_1
